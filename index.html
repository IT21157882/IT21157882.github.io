class Lottery
{
    private  int[] lotteryNumbers = new int[5];

    public Lottery()
    {
        int max = 9, min = 0, range = max - min + 1;
        
        for(int i = 0; i < this.lotteryNumbers.length; i++)
        {
            int rand = (int)(Math.random() * range) + min;
           
            boolean alreadyIn = false;
            
            for (int j = 0; j < this.lotteryNumbers.length; j++)
            {
                if(rand == this.lotteryNumbers[j])
                {
                    alreadyIn = true;
                    i--;
                    break;
                }
            }
            
            if(alreadyIn == false)
                lotteryNumbers[i] = rand;
        }
    }
    
    
    public int[]  lotteryPics(int[] arrayOfInteger)
    {
        int matchingTimes = 0;
        int matchingDigits[] = new int[]{-1, -1, -1 , -1, -1};
        
        for(int i = 0; i < this.lotteryNumbers.length; i++)
        {
            for (int j = 0; j < arrayOfInteger.length; j++)
            {
                if(this.lotteryNumbers[i] == arrayOfInteger[j])
                {
                    //System.out.println("ok");
                    matchingDigits[matchingTimes] = this.lotteryNumbers[i];
                    matchingTimes++;
                }
            }   
        }
        
        System.out.print("\nNumber of matching digits :   " + matchingTimes + " \n");
        
        return matchingDigits;
    }
    
    public void displayWiningNmbers()
    {
        System.out.print("\nLotteryr's NUmbers : ");
        for (int i = 0; i < this.lotteryNumbers.length; i++)
        {
            System.out.print(this.lotteryNumbers[i] + "  ");
        }
        
    }
}

class AlarmClock
{
        private int Hour;
        private int Minute;
        private int Second;

        public AlarmClock()
        {
                this.Hour = 12;
                this.Minute = 0;
                this.Second = 0;
        }
        
        public void InputAlarm()
        {
                Scanner reader = new Scanner(System.in);
                
                try 
                {
                        System.out.print("Hours  : ");
                        this.Hour = reader.nextInt();
                        
                         if(!(this.Hour <= 12 && this.Hour >= 0))
                                throw new HoursOutOfRange("\n\t * Hours should be in range to 0 to 12 *\n");
                        
                        System.out.print("Minute : ");
                        this.Minute = reader.nextInt();
                        
                        if(!(this.Minute < 60 && this.Minute >= 0))
                                throw new MinutesOutOfRange("\n\t * Minutes should be in range to 0 to 60 *\n");
                        
                        System.out.print("Second : ");
                        this.Second = reader.nextInt();
                                                
                        if(!(this.Second < 60 && this.Second >= 0))
                                throw new SecondsOutOfRange("\n\t * Seconds should be in range to 0 to 60 *\n");
                }
                catch (NumberFormatException e)
                {
                        System.out.println("\n\t ! You entered a non numeric value ! \n");
                }
                catch(HoursOutOfRange e)
                {
                        System.out.println(e.getMessage());
                        this.Hour = 12;
                }
                catch(MinutesOutOfRange e)
                {
                        System.out.println(e.getMessage());
                        this.Minute = 0;
                }
                catch(SecondsOutOfRange e)
                {
                        System.out.println(e.getMessage());
                        this.Second = 0;
                }
                
        }

        public void ShowAlarm() 
        {
                System.out.println("\nAlarmClock{ Hour=" + Hour + ", Minute=" + Minute + ", Second=" + Second + "}");
        }
        
        public void SetAlarm(int Hours, int Minutes, int seconds)
        {
                if(Hours <= 12 && Hours >= 0)
                        this.Hour = Hours;
                else
                        this.Hour = 0;
                
                if(Minutes < 60 && Minutes >= 0)
                        this.Minute = Minutes;
                else
                        this.Hour = 0;
                
                if(seconds < 60 && seconds >= 0 )
                        this.Second = seconds;
                else
                        this.Second = 0;
        }
        
              
}
